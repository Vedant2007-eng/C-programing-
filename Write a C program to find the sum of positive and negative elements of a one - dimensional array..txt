#include <stdio.h>
int main()
{
	int n,i,num;
	int posSum=0, negsum=0;
	scanf ("%d",&n);
	for (i=0;i<n;i++)
		{
			scanf("%d",&num);
			if(num>0)
				posSum+=num;
			else if(num<0)
				negsum+=num;
		}
	printf("%d\n",posSum);
	printf("%d\n",negsum);
	return 0;
}
